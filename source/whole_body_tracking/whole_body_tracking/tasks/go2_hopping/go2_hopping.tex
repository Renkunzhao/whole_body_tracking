\section{Commands}

Two command terms are sampled per env.

\paragraph{Hop ($\bm{c}_t$).} Per-env $(h^{*}_t, t^{*}_{\mathrm{stance},t})$
sampled uniformly from
\begin{equation}
    h^{*}_t \sim \mathcal{U}(0.1,\,0.6)\ \mathrm{m},
    \qquad
    t^{*}_{\mathrm{stance},t} \sim \mathcal{U}(0.1,\,0.5)\ \mathrm{s},
\end{equation}
with resampling interval drawn from $\mathcal{U}(6,\,12)\ \mathrm{s}$. The
commanded flight time $t^{*}_{\mathrm{flight},t} = 2\sqrt{2 h^{*}_t / g}$ and
cycle time $T_t$ are derived from these (Section~3.1).

\paragraph{Twist.} Per-env planar/yaw command sampled uniformly from
\begin{equation}
    v^{\mathrm{cmd}}_x,\,v^{\mathrm{cmd}}_y \sim \mathcal{U}(-0.5,\,0.5)\
    \mathrm{m/s},
    \qquad
    \omega^{\mathrm{cmd}}_z \sim \mathcal{U}(-0.5,\,0.5)\ \mathrm{rad/s},
\end{equation}
with resampling interval drawn from $\mathcal{U}(3,\,8)\ \mathrm{s}$, $50\%$
standing fraction, and a heading command driving $\omega^{\mathrm{cmd}}_z$ in
the remaining envs.

\section{Rewards}

IsaacLab applies the environment step time to each weighted reward term. For
this task,
\begin{equation}
    \Delta t = 4 \times 0.005 = 0.02\ \mathrm{s}.
\end{equation}
Let $n$ be the global environment step counter. The total reward used by the
environment is
\begin{equation}
\begin{aligned}
    R_t = \Delta t \Big(
        &2.0\,r^{\mathrm{phase}}_t
        + 140.0\,r^{\mathrm{air}}_t
        + w_{\mathrm{joint}}(n)\,r^{\mathrm{joint}}_t
        + w_{\mathrm{lin}}(n)\,r^{\mathrm{lin}}_t
        + w_{\mathrm{yaw}}(n)\,r^{\mathrm{yaw}}_t  \\
        &- 10^{-2}\,r^{\mathrm{action}}_t
        - 10.0\,r^{\mathrm{limit}}_t
    \Big).
\end{aligned}
\end{equation}
The curriculum weights are
\begin{equation}
    w_{\mathrm{joint}}(n) =
    \begin{cases}
        0, & n \le 1000 \times 24, \\
        0.5, & n > 1000 \times 24,
    \end{cases}
    \qquad
    w_{\mathrm{lin}}(n) =
    \begin{cases}
        0, & n \le 1000 \times 24, \\
        1.0, & n > 1000 \times 24,
    \end{cases}
    \qquad
    w_{\mathrm{yaw}}(n) =
    \begin{cases}
        0, & n \le 1000 \times 24, \\
        1.0, & n > 1000 \times 24.
    \end{cases}
\end{equation}
The curriculum terms are active in the current config; the three
phase-2 weights ramp from $0$ to their terminal values $(0.5,\,1.0,\,1.0)$
at step $n = 1000 \times 24$.

\subsection{Phase Contact}

The hopping phase is per-env, driven by the commanded
$(h^{*}_t, t^{*}_{\mathrm{stance},t})$ and the corresponding ballistic flight
time. With $g = 9.81\ \mathrm{m/s^2}$, the per-env cycle time is
\begin{equation}
    T_t = t^{*}_{\mathrm{stance},t} + 2\sqrt{\frac{2 h^{*}_t}{g}},
\end{equation}
and the normalized phase resets at every command resample:
\begin{equation}
    \phi_t =
    \operatorname{mod}
    \left(
        \frac{(k_t - k_{\mathrm{cycle\_start}})\,\Delta t}{T_t},
        1
    \right).
\end{equation}
The contact state of foot $i$ is
\begin{equation}
    c_{i,t} =
    \mathbb{I}
    \left[
        \max_{\tau \in \mathcal{H}}
        \left\lVert \bm{f}_{i,t-\tau} \right\rVert_2
        > f_{\mathrm{th}}
    \right],
    \qquad
    f_{\mathrm{th}} = 5.0\ \mathrm{N},
\end{equation}
where $\mathcal{H}$ is the contact-force history window. The desired contact
phase uses a per-env stance fraction
\begin{equation}
    \rho_{\mathrm{stance},t} = \frac{t^{*}_{\mathrm{stance},t}}{T_t},
    \qquad
    d_t =
    \mathbb{I}
    \left[
        \phi_t < \rho_{\mathrm{stance},t}
    \right].
\end{equation}
The raw phase-contact reward is
\begin{equation}
    r^{\mathrm{phase}}_t =
    d_t \prod_{i \in \mathcal{F}} c_{i,t}
    + (1 - d_t) \prod_{i \in \mathcal{F}} (1 - c_{i,t}),
\end{equation}
with weight $2.0$. Here $\mathcal{F}$ is the set of Go2 feet.

\subsection{Air-Time Tracking}

A pulse-on-landing reward ties each commanded flight window to a single
continuous air phase. Two gates suppress the local optima that
phase-contact alone cannot:
\begin{itemize}
    \item the pulse only fires on landings inside the commanded stance window
    ($\phi_t < \rho_{\mathrm{stance},t}$), so premature touchdowns during the
    commanded flight window yield zero;
    \item at most \emph{one} pulse fires per cycle. An ``armed'' flag $\alpha_t$
    starts true, is set false the moment the pulse fires, and re-arms on phase
    wrap (the start of the next cycle). This blocks the ``farm pulses by
    repeatedly mini-bouncing during stance'' exploit, since the second and
    later landings in the same cycle contribute zero.
\end{itemize}
With the commanded flight time
$t^{*}_{\mathrm{flight},t} = 2\sqrt{2 h^{*}_t / g}$ and
$\ell_t = \mathbb{I}[\text{air}\!\to\!\text{contact transition at } t]$
(``just landed''), the raw reward is
\begin{equation}
    r^{\mathrm{air}}_t =
    \ell_t \, d_t \, \alpha_t \,
    \exp\!\left(
        -\,
        \left(
            \frac{
                t^{\mathrm{last\_air}}_t - t^{*}_{\mathrm{flight},t}
            }{
                \sigma_{a}
            }
        \right)^2
    \right),
    \qquad
    \sigma_{a} = 0.3\ \mathrm{s},
\end{equation}
\begin{equation}
    \alpha_{t+1} =
    \big(
        \alpha_t \vee \mathbb{I}[\phi_{t+1} < \phi_t]
    \big)
    \wedge \neg(\ell_t d_t \alpha_t).
\end{equation}
Its weight is $140.0$.

\subsection{Joint Deviation (Phase-Dependent Exponential)}

Per-joint std values are selected by phase (stance vs. flight) using joint-name
regex maps. With $s_{j,t} = \sigma^{\mathrm{stance}}_j$ if $\phi_t <
\rho_{\mathrm{stance},t}$, and $\sigma^{\mathrm{flight}}_j$ otherwise, the raw
reward is
\begin{equation}
    r^{\mathrm{joint}}_t =
    \exp\!\left(
        -\,\frac{1}{|\mathcal{J}|}
        \sum_{j \in \mathcal{J}}
        \left(
            \frac{q_{j,t} - q^{0}_{j}}{s_{j,t}}
        \right)^{2}
    \right).
\end{equation}
The std maps used are
\begin{equation}
    \sigma^{\mathrm{stance}} =
    \begin{cases}
        0.5, & \text{hip} \\
        1.0, & \text{thigh} \\
        1.0, & \text{calf}
    \end{cases},
    \qquad
    \sigma^{\mathrm{flight}} =
    \begin{cases}
        0.5, & \text{hip} \\
        1.0, & \text{thigh} \\
        1.0, & \text{calf}
    \end{cases}.
\end{equation}
Curriculum sets the weight from $0$ to $0.5$ after $1000 \times 24$ steps.

\subsection{Linear Velocity Tracking}

The raw planar-velocity tracking reward is
\begin{equation}
    r^{\mathrm{lin}}_t =
    \exp\!\left(
        -\,
        \frac{
            \left\lVert
                \bm{v}^{\mathrm{cmd}}_{xy,t} - \bm{v}^{b}_{xy,t}
            \right\rVert_2^{2}
        }{
            \sigma_{v}^{2}
        }
    \right),
    \qquad
    \sigma_{v} = \sqrt{0.25}.
\end{equation}
The command term is \texttt{twist}. Curriculum sets the weight from $0$ to
$1.0$ after $1000 \times 24$ steps.

\subsection{Angular Velocity Tracking}

The raw yaw-rate tracking reward is
\begin{equation}
    r^{\mathrm{yaw}}_t =
    \exp
    \left(
        -
        \frac{
            \left(
                \omega^{\mathrm{cmd}}_{z,t}
                -
                \omega^{b}_{z,t}
            \right)^2
        }{
            \sigma_{\omega}^{2}
        }
    \right),
    \qquad
    \sigma_{\omega} = \sqrt{0.5}.
\end{equation}
The command term is \texttt{twist}. Curriculum sets the weight from $0$ to
$1.0$ after $1000 \times 24$ steps.

\subsection{Action Rate}

The raw action-rate penalty is
\begin{equation}
    r^{\mathrm{action}}_t =
    \sum_{m}
    \left(
        a_{m,t} - a_{m,t-1}
    \right)^2.
\end{equation}
Its weight is $-10^{-2}$.

\subsection{Joint Position Limits}

Let $\ell_j$ and $u_j$ be the soft lower and upper joint-position limits. With
$[x]_+ = \max(x,0)$, the raw joint-position-limit penalty is
\begin{equation}
    r^{\mathrm{limit}}_t =
    \sum_{j \in \mathcal{J}}
    \left(
        [\ell_j - q_{j,t}]_+
        +
        [q_{j,t} - u_j]_+
    \right).
\end{equation}
It uses all robot joints, and its weight is $-10.0$.

\section{Terminations}

An episode terminates when any of the following conditions hold:
\begin{itemize}
    \item \textbf{Base orientation:} the angle between the base $-z$ axis and
    gravity exceeds $1.0\ \mathrm{rad}$.
    \item \textbf{Non-foot contact:} any body other than the four feet
    (matched by regex \texttt{\^{}(?!.*\_foot\$).*}) registers contact force
    magnitude above $1.0\ \mathrm{N}$.
    \item \textbf{Time out:} the episode length reaches
    $20.0\ \mathrm{s}$ (counted as a time-out, not a failure).
\end{itemize}
